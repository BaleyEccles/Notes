b) Briefly describe the purpose of a slice (or equivalently a configurable logic block).
A configurable logic block is the fundamental component of FPGAs, it allow the implementation of logic and memory. The ug474 7Series CLB contains lookup tables, memory, shift registers, multiplexer and few more things.

The output of the CLB will be physically connected to a wire, the switch matrix will be used to connect this wire back to the physical input of the CLB.

module coin_toss(input clk, input TOSS, input RST, output reg[1:0] Z, output reg [2:0] state);
   // All possible states:
   parameter idle = 3'b00;
   parameter heads = 3'b01;
   parameter tails = 3'b10;
   
   
   always @(posedge clk) begin
      // State machine:
      if (TOSS) begin
         case(state)
           idle: state <= heads;
           heads: state <= tails;
           tails: state <= heads;
         endcase
      end
      if (RST) begin
         case(state)
           idle: state <= idle;
           heads: state <= idle;
           tails: state <= idle;
         endcase
      end

      // Output:
      case(state)
        idle: Z <= 2'b00;
        heads: Z <= 2'b01;
        tails: Z <= 2'b10;
      endcase
   end
   
   initial begin
      state <= idle;
      Z <= 2'b00;
   end
endmodule

module fib(input clk, input en, input rst, input [3:0] n, output reg odd, output reg even, output reg [7:0] data_out);
   integer i;
   integer next;
   integer a;
   integer b;
   
   integer ones_count;
   
   always @(posedge clk) begin
      if (en == 1) begin // a
         data_out = 8'bZ;
      end else begin
         // c
         case (n)
           4'b0000: data_out = 8'b00000000;
           4'b0001: data_out = 8'b00000001;
           default: begin
              next <= 0;
              a <= 0;
              b <= 1;
              for (i = 0; i < n; i = i + 1) begin
                 if (next >= 255) data_out = 8'hFF;
                 else data_out = next[7:0];
                 next = a + b;
                 a = b;
                 b = next;
              end
           end
         endcase
         ones_count = 0;
         for (i = 0; i < 8; i = i + 1) begin
            if (data_out[i] == 1'b1) ones_count <= ones_count + 1;
         end
         if (ones_count % 2 == 0) even = 1'b1;
         if (ones_count % 2 != 0) even = 1'b0;
         odd = ~even;
      end
      if (rst == 0 && en != 1) begin // b
         data_out <= 8'b0;
      end
   end

endmodule
